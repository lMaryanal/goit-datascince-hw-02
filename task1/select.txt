--Отримати всі завдання певного користувача.
SELECT title
FROM tasks
WHERE user_id = 2;

--Вибрати завдання за певним статусом.
SELECT title
FROM tasks
WHERE status_id IN(SELECT id 
FROM status
WHERE name LIKE 'new'
);

--Оновити статус конкретного завдання.
UPDATE tasks SET status_id = 2 WHERE id = 1;

--Отримати список користувачів, які не мають жодного завдання.
SELECT fullname
FROM users 
WHERE id NOT IN (SELECT user_id
FROM tasks 
);

--Додати нове завдання для конкретного користувача.
INSERT INTO tasks (title, user_id, status_id)
VALUES ('new task', 5, 1);

--Отримати всі завдання, які ще не завершено. 
SELECT title
FROM tasks
WHERE status_id != 3; 

--Видалити конкретне завдання.
DELETE FROM tasks WHERE id = 40;

--Знайти користувачів з певною електронною поштою.
SELECT fullname
FROM users
WHERE email LIKE '%com%';

--Оновити ім'я користувача.
UPDATE users SET fullname = 'Newname' WHERE id = 10;

--Отримати кількість завдань для кожного статусу. 
SELECT COUNT(id) as id, status_id
FROM tasks
GROUP BY status_id;

--Отримати завдання, які призначені користувачам з певною доменною частиною електронної пошти.
SELECT t.title
FROM tasks as t
JOIN users AS u ON u.id = t.user_id
WHERE u.email LIKE'%@example.com'; 

--Отримати список завдань, що не мають опису.
SELECT title
FROM tasks
WHERE description is NULL;

--Вибрати користувачів та їхні завдання, які є у статусі 'in progress'.
SELECT u.fullname, t.title 
FROM tasks as t
INNER JOIN users AS u ON u.id = t.user_id
WHERE t.status_id = 2; 

--Отримати користувачів та кількість їхніх завдань.
SELECT u.fullname, COUNT(t.id) as tasks
FROM users as u
LEFT JOIN tasks AS t ON u.id = t.user_id
GROUP BY u.fullname;
